<?php
function errorBox($color,$position,$title,$text) {
	if ($position=="inline") {
		echo "<div style='width:500px; position: relative; text-align:center; margin-left:auto; margin-right:auto;' class='" . $color . "Error' id='" . $idName . "'>";

		echo "<p class='error'>" . $text . "</p>";
		echo "</div>";

		echo "<div style='";
		if ($title=="hide") { echo "display:none; "; };
 		echo "position:fixed; z-index:8; float:top; right:-5px; top:-5px; text-align:center; vertical-align:middle; padding:8px;' class='" . $color . "Error'";

		} else {
			echo ">";

		}
		echo "</div>";
	}
}
